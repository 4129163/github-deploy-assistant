```javascript
// src/services/process-manager.js
class ProcessManager {
  // 启动项目进程
  async startProject(projectId, command, cwd) {
    const proc = spawn(command, {
      cwd,
      stdio: ['pipe', 'pipe', 'pipe'],
      shell: true
    });
    
    // 进程监控
    this.monitorProcess(proc, projectId);
    
    // 状态持久化
    await this.saveProcessState(projectId, proc.pid, 'running');
    
    return proc;
  }
  
  // 健康检查
  async healthCheck(projectId) {
    const proc = this.getProcess(projectId);
    if (!proc) return { status: 'stopped' };
    
    // 检查进程是否存活
    try {
      process.kill(proc.pid, 0); // 发送0信号检查存活
      return { status: 'running', pid: proc.pid };
    } catch (e) {
      return { status: 'stopped', error: '进程已终止' };
    }
  }
}
```

```javascript
async function scanLocalProjects(rootDir) {
  const projects = [];
  
  // 使用广度优先搜索
  const dirs = await fs.readdir(rootDir, { withFileTypes: true });
  
  for (const dir of dirs) {
    if (!dir.isDirectory()) continue;
    
    const dirPath = path.join(rootDir, dir.name);
    
    // 检测项目类型
    const projectType = await detectProjectType(dirPath);
    
    if (projectType) {
      // 解析Git信息（如果有）
      const gitInfo = await parseGitInfo(dirPath);
      
      projects.push({
        name: dir.name,
        path: dirPath,
        type: projectType,
        git: gitInfo,
        size: await getDirSize(dirPath)
      });
    }
  }
  
  return projects;
}
```

### 3.2 项目类型检测
```javascript
async function detectProjectType(dirPath) {
  const files = await fs.readdir(dirPath);
  
  // 检查关键文件
  if (files.includes('package.json')) return 'nodejs';
  if (files.includes('requirements.txt')) return 'python';
  if (files.includes('Dockerfile')) return 'docker';
  if (files.includes('go.mod')) return 'go';
  if (files.includes('Cargo.toml')) return 'rust';
  if (files.includes('pom.xml')) return 'java-maven';
  if (files.includes('build.gradle')) return 'java-gradle';
  
  // 检查.git目录（可能是任何项目）
  if (await fs.pathExists(path.join(dirPath, '.git'))) {
    return 'git-repo';
  }
  
  return null;
}
```

```javascript
class ResourceMonitor {
  constructor() {
    this.metrics = {
      cpu: [],

    };
    
    // 每5秒收集一次数据
    this.interval = setInterval(() => this.collectMetrics(), 5000);

    const timestamp = Date.now();
    
    // CPU使用率
    const cpuUsage = await getCpuUsage();
    this.metrics.cpu.push({ timestamp, value: cpuUsage });
    
    // 内存使用
    const memoryUsage = await getMemoryUsage();
    this.metrics.memory.push({ timestamp, value: memoryUsage });
    
    // 保持最近1小时数据
    this.trimOldData();
  }
  
  trimOldData() {
    const oneHourAgo = Date.now() - 3600000;
    
    for (const key in this.metrics) {
      this.metrics[key] = this.metrics[key].filter(
        point => point.timestamp > oneHourAgo
      );
    }
  }
  
  getMetrics() {
    return {
      cpu: this.metrics.cpu,
      memory: this.metrics.memory,
      disk: this.metrics.disk,
      network: this.metrics.network,
      last_updated: Date.now()
    };
  }
}
```

```javascript
    for (const [key, value] of this.cache.entries()) {
      if (now - value.timestamp > this.ttl) {
        this.cache.delete(key);
      }
    }
  }
}
```

### 错误处理

#### 1. 统一错误处理
```javascript
class ErrorHandler {
  static handle(error, context = {}) {
    const errorId = uuid.v4();
    const errorInfo = {
      id: errorId,
      timestamp: new Date(),
      message: error.message,
      stack: error.stack,
      context,
      type: this.classifyError(error)
    };
    
    // 记录错误日志
    logger.error('System error:', errorInfo);
    
    // 发送错误通知（如果配置了）
    if (process.env.ERROR_NOTIFICATION_URL) {
      this.sendNotification(errorInfo);
    }
    
    // 返回用户友好的错误信息
    return {
      error: true,
      error_id: errorId,
      message: this.getUserFriendlyMessage(error),
      ...(process.env.NODE_ENV === 'development' ? { details: errorInfo } : {})
    };
  }
  
  static classifyError(error) {
    if (error.code === 'ENOENT') return 'file_not_found';
    if (error.code === 'EACCES') return 'permission_denied';
    if (error.code === 'ECONNREFUSED') return 'connection_refused';
    if (error.message.includes('timeout')) return 'timeout';
    if (error.message.includes('memory')) return 'out_of_memory';
    return 'unknown';
  }
  
  static getUserFriendlyMessage(error) {
    const type = this.classifyError(error);
    
    const messages = {
      file_not_found: '文件不存在，请检查路径是否正确',
      permission_denied: '权限不足，请检查文件权限',
      connection_refused: '连接被拒绝，请检查网络连接',
      timeout: '操作超时，请稍后重试',
      out_of_memory: '内存不足，请关闭其他程序',
      unknown: '发生未知错误，请稍后重试'
    };
    
    return messages[type] || messages.unknown;
  }
}
```

```javascript
class AlertManager {
  constructor() {
    this.alertRules = this.loadAlertRules();
    this.notificationChannels = this.loadChannels();
  }
  
  async checkAndAlert(metrics) {
    for (const rule of this.alertRules) {
      const shouldAlert = this.evaluateRule(rule, metrics);
      
      if (shouldAlert) {
        await this.sendAlert(rule, metrics);
      }
    }
  }
  
  evaluateRule(rule, metrics) {
    const value = metrics[rule.metric];
    
    switch (rule.condition) {
      case 'gt': return value > rule.threshold;
      case 'lt': return value < rule.threshold;
      case 'eq': return value === rule.threshold;
      case 'neq': return value !== rule.threshold;
      default: return false;
    }
  }
  
  async sendAlert(rule, metrics) {
    const alert = {
      id: uuid.v4(),
      rule: rule.name,
      metric: rule.metric,
      value: metrics[rule.metric],
      threshold: rule.threshold,
      condition: rule.condition,
      timestamp: new Date(),
      severity: rule.severity
    };
    
    // 发送到所有配置的通道
    for (const channel of this.notificationChannels) {
      await this.sendToChannel(channel, alert);
    }
    
    // 记录警报
    await AlertDB.create(alert);
  }
}
```
